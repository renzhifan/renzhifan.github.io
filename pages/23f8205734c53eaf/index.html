cd SMProxy
bin/SMProxy start

{
  "database": {
    "account": {
      "db_user": { # 我就是错在这里了  之前是root  我忘记修改 导致报错 改完之后即可正常运行

        "password": "password"
      }
    },
    "serverInfo": {
      "server1": {
        "write": {
          "host": ["127.0.0.1"],
          "port": 3306,
          "timeout": 2,
          "account": "zhifan"
        },
        "read": {
          "host": ["127.0.0.1"],
